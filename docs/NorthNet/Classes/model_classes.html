146    def load_conditions(self, data_report):
147        """
148        Load conditions details into ModelWriter attributes to allow compilation of
149        experimental conditions into the model.
150
151        For now, this method will assume that values are in base SI units (e.g.
152        M, not mM), and the keys to conditions should follow some relatively strict
153        patterns.
154
155        "reactor_volume": gives the reactor volume in L
156        "{}/ M": gives the concentration of an inlet in M.
157        "{}_flow_{}", not containing "time": gives the flow rate of an input,
158        in L/ s
159        "{}_flow_time_{}": (contains "flow" and "time") gives the time axis of
160        the flow profiles.
161
162        Parameters
163        ----------
164        data_report: NorthNet.Classes.DataReport
165
166        Returns
167        -------
168        None
169        """
170
171        # Get the series values from the conditions
172        self.time = data_report.series_values.copy()
173
174        # Extract flow input information
175        for condition in data_report.conditions:
176            # Get the reactor volume
177            if "reactor_volume" in condition:
178                self.reactor_volume = data_report.conditions[condition]
179
180            # Get the input concentrations
181            elif "/ M" in condition:
182                smiles = condition.split("/")[0]
183                self.inputs[smiles] = data_report.conditions[condition]
184
185            # Get the flow profile time
186            elif "time" in condition and "flow" in condition:
187                self.flow_profile_time = np.array(data_report.conditions[condition])
188
189            # Get flow profile
190            elif "flow" in condition and not "time" in condition:
191                smiles = condition.split("_")[0]
192                self.flow_profiles[smiles] = np.array(data_report.conditions[condition])
193
194        # Get the total flow rate of all the inputs
195        self.sigma_flow = np.zeros(len(self.flow_profile_time))
196        for flow in self.flow_profiles:
197            self.sigma_flow += self.flow_profiles[flow]
198
199        # update the network to reflect the inputs and outputs implied by the
200        # conditions
201        for inp in self.inputs:
202            input = Classes.ReactionInput(f"{inp}_#0>>{inp}")
203            self.network.add_input_process(input)
204
205        # No information about multiple outputs for now.
206        # Connect a single output to all of the compounds in the network
207        for comp in self.network.NetworkCompounds:
208            output = Classes.ReactionOutput(f"{comp}>>#0")
209            self.network.add_output_process(output)

211    def write_flow_profile_text(self, indentation=""):
212        """
213        Write a series of concentration profiles for each input and a total
214        flow rate profile.
215
216        Parameters
217        ----------
218        indentation: str
219
220        Returns
221        -------
222        text: str
223        """
224
225        if len(self.flow_profiles) == 0:
226            # No flow profile information
227            return ""
228
229        network = self.network
230
231        input_concentrations = np.zeros(
232            (len(self.flow_profiles), len(self.flow_profile_time))
233        )
234
235        for c, flow in enumerate(self.inflows):
236
237            compound = flow.split("_")[0]
238
239            if compound in self.inputs:
240                conc = self.inputs[compound]
241                flow_rate = self.flow_profiles[flow.split("_")[0]]
242                conc_profile = conc * flow_rate / self.reactor_volume
243                input_concentrations[c] = conc_profile
244
245        total_flows = np.zeros((len(self.outflows), len(self.sigma_flow)))
246
247        # assume that the output flow rates are equally partitioned between the
248        # output channels.
249        partitioned_flow = self.sigma_flow / len(self.outflows)
250        for c, out in enumerate(self.outflows):
251            total_flows[c] = partitioned_flow / self.reactor_volume
252
253        # Write concentration array
254        text = f"{indentation}F_in = np.array(\n"
255        text += indentation
256        array_text = np.array2string(
257            input_concentrations,
258            formatter={"float_kind": lambda x: "%.9f" % x},
259            separator=",",
260            threshold=np.inf,
261        )
262        text += array_text.replace("\n", f"\n{indentation}")
263        text += f"{indentation})\n\n"
264
265        # Write flow profile time axis
266        text += f"{indentation}flow_time = np.array(\n"
267        text += indentation
268        array_text = np.array2string(
269            self.flow_profile_time,
270            formatter={"float_kind": lambda x: "%.9f" % x},
271            separator=",",
272            threshold=np.inf,
273        )
274        text += array_text.replace("\n", f"\n{indentation}")
275        text += f"{indentation})\n\n"
276
277        # Write total flow rate.
278        text += f"{indentation}total_flow = np.array(\n"
279        text += indentation
280        array_text = np.array2string(
281            total_flows,
282            formatter={"float_kind": lambda x: "%.9f" % x},
283            separator=",",
284            threshold=np.inf,
285        )
286        text += array_text.replace("\n", f"\n{indentation}")
287        text += f"{indentation})\n"
288
289        return text

291    def write_model_equation_text(self):
292        """
293        Writes models as equations with variables which refer to indices of
294        arrays:
295
296        P: 1D array of len(self.network.NetworkCompounds)
297            Stores the product state if the system following calculation
298        S: 1D array of len(self.network.NetworkCompounds)
299            Stores the initial state of the system following calculation
300            (see self.get_network_tokens())
301        k: 1D array of len(self.network.NetworkReactions)
302            rate constants arranged in standardised order.
303            (see self.get_network_tokens())
304
305        Includes output flow terms for all compounds.
306
307        Parameters
308        ----------
309        None
310
311        Returns
312        -------
313        eq_lines: list
314            List of rate equations in text form.
315        """
316
317        network = self.network
318
319        compounds = [*network.NetworkCompounds]
320
321        eq_lines = []
322
323        for count, compound in enumerate(compounds):
324            line_text = f"P[{count}] = "
325            for i in network.NetworkCompounds[compound].In:
326                if i in network.InputProcesses:
327                    input_id = network.InputProcesses[i].InputID
328                    input_conc = self.inflows[input_id]
329                    ki = f"+{input_conc}"
330                    line_text += ki
331                else:
332                    reactants = network.NetworkReactions[i].Reactants
333
334                    ki = f"+{self.rate_constants[i]}*"
335
336                    if len(reactants) == 0:
337                        specs = ""
338                    else:
339                        specs = "*".join([self.species[x] for x in reactants])
340
341                    line_text += f"{ki}{specs}"
342
343            for out in network.NetworkCompounds[compound].Out:
344                if out in network.OutputProcesses:
345                    output_process = network.OutputProcesses[out]
346                    out_compound = output_process.OutputCompound
347                    outlet = output_process.OutputID
348                    out_flow = self.outflows[outlet]
349                    reactor_conc = self.species[out_compound]
350                    ki = f"-{reactor_conc}*{out_flow}"
351                    line_text += ki
352                else:
353                    ki = f"-{self.rate_constants[out]}*"
354                    reactants = [
355                        self.species[x] for x in network.NetworkReactions[out].Reactants
356                    ]
357                    specs = "*".join(reactants)
358                    line_text += f"{ki}{specs}"
359
360            eq_lines.append(line_text)
361
362        return eq_lines

364    def write_variables_text(self):
365        """
366        Write model variables as strings stored in a list
367
368        Parameters
369        ----------
370        None
371
372        Returns
373        -------
374        lines: list[str]
375        """
376
377        get_index = lambda x: int(x[x.find("[") + 1 : x.find("]")])
378        lines = []
379
380        lines.append("species = {")
381        for k in self.species:
382            idx = get_index(self.species[k])
383            lines.append(f"'{k}':{idx},")
384        lines.append("}")
385
386        lines.append("")
387        lines.append("reactions = {")
388        for k in self.rate_constants:
389            idx = get_index(self.rate_constants[k])
390            lines.append(f"'{k}':{idx},")
391        lines.append("}")
392
393        lines.append("")
394        lines.append("inputs = {")
395        for k in self.inputs:
396            idx = self.inputs[k]
397            lines.append(f"'{k}':{idx},")
398        lines.append("}")
399        lines.append("")
400
401        lines.append("k = np.zeros(max(reactions.values())+1) # rate constants")
402        lines.append("")
403        lines.append("S = np.zeros(len(species)) # initial concentrations")
404
405        return lines

407    def write_to_module_text(self, numba_decoration=None):
408        """
409
410        Parameters
411        ----------
412        numba_decoration: str or None
413            Which numba decoration to add.
414            "jit": Decoration for just in time compilation.
415            "compile": Decoration for ahead of time compilation
416            None: No decoration.
417
418        Returns
419        -------
420        text: str
421            The module text.
422        """
423
424        flow_profile_text = self.write_flow_profile_text(indentation="    ")
425
426        model_text = self.write_model_equation_text()
427
428        nf64 = "numba.float64"
429        numba_dec = ""
430        numba_dec += f"@numba.jit({nf64}[:]({nf64},{nf64}[:],{nf64}[:]),\n"
431        numba_dec += "\tlocals="
432
433        if flow_profile_text == "":
434            numba_dec += f"{{'P': {nf64}[:]}}"
435        else:
436            numba_dec += f"{{'P': {nf64}[:],'F': {nf64}[:,:],'I':{nf64}[:]}}"
437
438        numba_dec += ",nopython=True)"
439
440        lines = ["import numpy as np"]
441        if numba_decoration == "jit":
442            lines.append("import numba\n")
443            lines.append("")
444            lines.append(numba_dec)
445
446        if numba_decoration == "compile":
447            lines.append("import numba")
448            lines.append("from numba.pycc import CC\n")
449            mod_name = self.name
450            if self.name == "":
451                mod_name = "model_func"
452            lines.append(f"cc = CC('{mod_name}')\n")
453            lines.append(
454                '@cc.export("model_func", "float64[:](float64,float64[:],float64[:])")'
455            )
456
457        lines.append("def model_function(time, S, k):")
458        lines.append("")
459        lines.append("    P = np.zeros(len(S))")
460
461        if flow_profile_text != "":
462            lines.append("")
463            lines.append(flow_profile_text)
464            lines.append("    i = np.abs(flow_time - time).argmin()")
465            lines.append("")
466
467        for m_text in model_text:
468            lines.append(f"    {m_text}")
469
470        lines.append("    P *= time")
471        lines.append("")
472        lines.append("    return P")
473        lines.append("")
474        lines.append("def wrapper_function(time, S, k):")
475        lines.append("    return model_function(time, S, k)")
476        lines.append("")
477
478        lines.extend(self.write_variables_text())
479
480        lines.append("")
481        if numba_decoration == "compile":
482            lines.append('if __name__ == "__main__":')
483            lines.append("    cc.compile()")
484
485        text = "\n".join(lines)
486
487        return text

489    def write_model_matrix_text(self):
490        """
491        Prototype for writing the model as an array.
492
493        Parameters
494        ----------
495
496        Returns
497        -------
498        mat_text: str
499            Rate equations in numpy matrix form.
500        """
501        compounds = [*self.network.NetworkCompounds]
502
503        species = self.species
504        rate_consts = self.rate_constants
505        inflows = self.inputs
506
507        ratemat = [["0" for _ in species] for _ in species]
508
509        for compound in compounds:
510            ind1 = compounds.index(compound)
511
512            """outgoing reactions"""
513            for i in self.network.NetworkCompounds[compound].In:
514
515                token = ""
516                ind2 = ind1
517                reacs = self.network.NetworkReactions[i].Reactants
518                ki = rate_consts[i]
519
520                if len(reacs) == 0:
521                    token = f"(+{ki}*{inflows[compound]})/{species[compound]}"
522                    ind2 = ind1
523
524                if len(reacs) == 1:
525                    n2 = reacs[0]
526                    ind2 = compounds.index(n2)
527                    token = "+" + ki
528
529                if len(reacs) == 2:
530                    n2 = reacs[0]
531                    n3 = reacs[1]
532                    ind2 = compounds.index(n3)
533                    token = "+" + ki + "*" + species[n2]
534
535                if len(reacs) == 3:
536                    n2 = reacs[0]
537                    n3 = reacs[1]
538                    n4 = reacs[2]
539                    ind2 = compounds.index(n3)
540                    token = "+" + ki + "*" + species[n2] + "*" + species[n4]
541
542                ratemat[ind1][ind2] += token
543
544            for out in self.network.NetworkCompounds[compound].Out:
545
546                token = ""
547                ind2 = ind1
548                reacs = self.network.NetworkReactions[out].Reactants
549                ki = rate_consts[out]
550
551                if len(reacs) == 1:
552                    token = "-" + ki
553                    ind2 = ind1
554
555                if len(reacs) == 2:
556                    z = reacs[:]
557                    z.remove(compound)
558                    n2 = z[0]
559                    ind2 = compounds.index(n2)
560                    token = "-" + ki + "*" + species[compound]
561
562                if len(reacs) == 3:
563                    z = reacs[:]
564                    z.remove(compound)
565                    n2 = z[0]
566                    n3 = z[1]
567                    ind2 = compounds.index(n2)
568                    token = "-" + ki + "*" + species[compound] + "*" + species[n3]
569
570                ratemat[ind1][ind2] += token
571
572        mat_text = "["
573        for element in ratemat:
574            mat_text += "[" + ",".join(element) + "],\n"
575
576        mat_text = mat_text.strip(",\n") + "]"
577
578        return mat_text

292    def write_model_equation_text(self):
293        """
294        Writes models as equations with variables which refer to indices of
295        arrays:
296
297        P: 1D array of len(self.network.NetworkCompounds)
298            Stores the product state if the system following calculation
299        S: 1D array of len(self.network.NetworkCompounds)
300            Stores the initial state of the system following calculation
301            (see self.get_network_tokens())
302        k: 1D array of len(self.network.NetworkReactions)
303            rate constants arranged in standardised order.
304            (see self.get_network_tokens())
305
306        Includes output flow terms for all compounds.
307
308        Parameters
309        ----------
310        None
311
312        Returns
313        -------

315            List of rate equations in text form.
316        """
317
318        network = self.network
319
320        compounds = [*network.NetworkCompounds]
321
322        eq_lines = []
323
324        for count, compound in enumerate(compounds):
325            line_text = f"P[{count}] = "
326            for i in network.NetworkCompounds[compound].In:
327                if i in network.InputProcesses:
328                    input_id = network.InputProcesses[i].InputID
329                    input_conc = self.inflows[input_id]
330                    ki = f"+{input_conc}"
331                    line_text += ki
332                else:
333                    reactants = network.NetworkReactions[i].Reactants
334
335                    ki = f"+{self.rate_constants[i]}*"
336
337                    if len(reactants) == 0:
338                        specs = ""
339                    else:
340                        specs = "*".join([self.species[x] for x in reactants])
341
342                    line_text += f"{ki}{specs}"
343
344            for out in network.NetworkCompounds[compound].Out:
345                if out in network.OutputProcesses:
346                    output_process = network.OutputProcesses[out]
347                    out_compound = output_process.OutputCompound
348                    outlet = output_process.OutputID
349                    out_flow = self.outflows[outlet]
350                    reactor_conc = self.species[out_compound]
351                    ki = f"-{reactor_conc}*{out_flow}"
352                    line_text += ki
353                else:
354                    ki = f"-{self.rate_constants[out]}*"
355                    reactants = [
356                        self.species[x] for x in network.NetworkReactions[out].Reactants
357                    ]
358                    specs = "*".join(reactants)
359                    line_text += f"{ki}{specs}"
360
361            eq_lines.append(line_text)
362
363        return eq_lines
