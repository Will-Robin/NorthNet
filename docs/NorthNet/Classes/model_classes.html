114        # shorthand for the network

116

136        inflow_rates = {k: f"F_in[{c},i]" for c, k in enumerate(input_ids)}

145    def load_conditions(self, data_report):
146        """
147        Load conditions details into ModelWriter attributes to allow compilation of
148        experimental conditions into the model.
149
150        For now, this method will assume that values are in base SI units (e.g.
151        M, not mM), and the keys to conditions should follow some relatively strict
152        patterns.
153
154        "reactor_volume": gives the reactor volume in L
155        "{}/ M": gives the concentration of an inlet in M.
156        "{}_flow_{}", not containing "time": gives the flow rate of an input,
157        in L/ s
158        "{}_flow_time_{}": (contains "flow" and "time") gives the time axis of
159        the flow profiles.
160
161        Parameters
162        ----------
163        data_report: NorthNet.Classes.DataReport
164
165        Returns
166        -------
167        None
168        """
169
170        # Get the series values from the conditions
171        self.time = data_report.series_values.copy()
172
173        # Extract flow input information
174        for condition in data_report.conditions:
175            # Get the reactor volume
176            if "reactor_volume" in condition:
177                self.reactor_volume = data_report.conditions[condition]
178
179            # Get the input concentrations
180            elif "/ M" in condition:
181                smiles = condition.split("/")[0]
182                self.inputs[smiles] = data_report.conditions[condition]
183
184            # Get the flow profile time
185            elif "time" in condition and "flow" in condition:
186                self.flow_profile_time = np.array(data_report.conditions[condition])
187
188            # Get flow profile
189            elif "flow" in condition and not "time" in condition:
190                smiles = condition.split("_")[0]
191                self.flow_profiles[smiles] = np.array(data_report.conditions[condition])
192
193        # Get the total flow rate of all the inputs
194        self.sigma_flow = np.zeros(len(self.flow_profile_time))
195        for flow in self.flow_profiles:
196            self.sigma_flow += self.flow_profiles[flow]
197
198        # update the network to reflect the inputs and outputs implied by the
199        # conditions
200        for inp in self.inputs:
201            input = Classes.ReactionInput(f"{inp}_#0>>{inp}")
202            self.network.add_input_process(input)
203
204        # No information about multiple outputs for now.
205        # Connect a single output to all of the compounds in the network
206        for comp in self.network.NetworkCompounds:
207            output = Classes.ReactionOutput(f"{comp}>>#0")
208            self.network.add_output_process(output)

302    def write_model_matrix_text(self):
303        """
304        Prototype for writing the model as an array.
305
306        Parameters
307        ----------
308
309        Returns
310        -------
311        mat_text: str
312            Rate equations in numpy matrix form.
313        """
314        compounds = [*self.network.NetworkCompounds]
315
316        species = self.species
317        rate_consts = self.rate_constants
318        inflows = self.inputs
319
320        ratemat = [["0" for _ in species] for _ in species]
321
322        for compound in compounds:
323            ind1 = compounds.index(compound)
324
325            """outgoing reactions"""
326            for i in self.network.NetworkCompounds[compound].In:
327
328                token = ""
329                ind2 = ind1
330                reacs = self.network.NetworkReactions[i].Reactants
331                ki = rate_consts[i]
332
333                if len(reacs) == 0:
334                    token = f"(+{ki}*{inflows[compound]})/{species[compound]}"
335                    ind2 = ind1
336
337                if len(reacs) == 1:
338                    n2 = reacs[0]
339                    ind2 = compounds.index(n2)
340                    token = "+" + ki
341
342                if len(reacs) == 2:
343                    n2 = reacs[0]
344                    n3 = reacs[1]
345                    ind2 = compounds.index(n3)
346                    token = "+" + ki + "*" + species[n2]
347
348                if len(reacs) == 3:
349                    n2 = reacs[0]
350                    n3 = reacs[1]
351                    n4 = reacs[2]
352                    ind2 = compounds.index(n3)
353                    token = "+" + ki + "*" + species[n2] + "*" + species[n4]
354
355                ratemat[ind1][ind2] += token
356
357            for out in self.network.NetworkCompounds[compound].Out:
358
359                token = ""
360                ind2 = ind1
361                reacs = self.network.NetworkReactions[out].Reactants
362                ki = rate_consts[out]
363
364                if len(reacs) == 1:
365                    token = "-" + ki
366                    ind2 = ind1
367
368                if len(reacs) == 2:
369                    z = reacs[:]
370                    z.remove(compound)
371                    n2 = z[0]
372                    ind2 = compounds.index(n2)
373                    token = "-" + ki + "*" + species[compound]
374
375                if len(reacs) == 3:
376                    z = reacs[:]
377                    z.remove(compound)
378                    n2 = z[0]
379                    n3 = z[1]
380                    ind2 = compounds.index(n2)
381                    token = "-" + ki + "*" + species[compound] + "*" + species[n3]
382
383                ratemat[ind1][ind2] += token
384
385        mat_text = "["
386        for element in ratemat:
387            mat_text += "[" + ",".join(element) + "],\n"
388
389        mat_text = mat_text.strip(",\n") + "]"
390
391        return mat_text

def load_conditions(self, data_report) 
146    def load_conditions(self, data_report):
147        """
148        Load conditions details into ModelWriter attributes to allow compilation of
149        experimental conditions into the model.
150
151        For now, this method will assume that values are in base SI units (e.g.
152        M, not mM), and the keys to conditions should follow some relatively strict
153        patterns.
154
155        "reactor_volume": gives the reactor volume in L
156        "{}/ M": gives the concentration of an inlet in M.
157        "{}_flow_{}", not containing "time": gives the flow rate of an input,
158        in L/ s
159        "{}_flow_time_{}": (contains "flow" and "time") gives the time axis of
160        the flow profiles.
161
162        Parameters
163        ----------
164        data_report: NorthNet.Classes.DataReport
165
166        Returns
167        -------
168        None
169        """
170
171        # Get the series values from the conditions
172        self.time = data_report.series_values.copy()
173
174        # Extract flow input information
175        for condition in data_report.conditions:
176            # Get the reactor volume
177            if "reactor_volume" in condition:
178                self.reactor_volume = data_report.conditions[condition]
179
180            # Get the input concentrations
181            elif "/ M" in condition:
182                smiles = condition.split("/")[0]
183                self.inputs[smiles] = data_report.conditions[condition]
184
185            # Get the flow profile time
186            elif "time" in condition and "flow" in condition:
187                self.flow_profile_time = np.array(data_report.conditions[condition])
188
189            # Get flow profile
190            elif "flow" in condition and not "time" in condition:
191                smiles = condition.split("_")[0]
192                self.flow_profiles[smiles] = np.array(data_report.conditions[condition])
193
194        # Get the total flow rate of all the inputs
195        self.sigma_flow = np.zeros(len(self.flow_profile_time))
196        for flow in self.flow_profiles:
197            self.sigma_flow += self.flow_profiles[flow]
198
199        # update the network to reflect the inputs and outputs implied by the
200        # conditions
201        for inp in self.inputs:
202            input = Classes.ReactionInput(f"{inp}_#0>>{inp}")
203            self.network.add_input_process(input)
204
205        # No information about multiple outputs for now.
206        # Connect a single output to all of the compounds in the network
207        for comp in self.network.NetworkCompounds:
208            output = Classes.ReactionOutput(f"{comp}>>#0")
209            self.network.add_output_process(output)

322

324            ind1 = compounds.index(compound)
325
326            """outgoing reactions"""

328
329                token = ""

331                reacs = self.network.NetworkReactions[i].Reactants
332                ki = rate_consts[i]
333
334                if len(reacs) == 0:
335                    token = f"(+{ki}*{inflows[compound]})/{species[compound]}"
336                    ind2 = ind1
337
338                if len(reacs) == 1:
339                    n2 = reacs[0]
340                    ind2 = compounds.index(n2)
341                    token = "+" + ki
342
343                if len(reacs) == 2:
344                    n2 = reacs[0]
345                    n3 = reacs[1]
346                    ind2 = compounds.index(n3)
347                    token = "+" + ki + "*" + species[n2]
348
349                if len(reacs) == 3:
350                    n2 = reacs[0]
351                    n3 = reacs[1]
352                    n4 = reacs[2]
353                    ind2 = compounds.index(n3)
354                    token = "+" + ki + "*" + species[n2] + "*" + species[n4]
355
356                ratemat[ind1][ind2] += token
357
358            for out in self.network.NetworkCompounds[compound].Out:
359
360                token = ""
361                ind2 = ind1
362                reacs = self.network.NetworkReactions[out].Reactants
363                ki = rate_consts[out]
364
365                if len(reacs) == 1:
366                    token = "-" + ki
367                    ind2 = ind1
368
369                if len(reacs) == 2:
370                    z = reacs[:]
371                    z.remove(compound)
372                    n2 = z[0]
373                    ind2 = compounds.index(n2)
374                    token = "-" + ki + "*" + species[compound]
375
376                if len(reacs) == 3:
