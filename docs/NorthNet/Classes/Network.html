179    def add_reaction(self, reaction):
180        """
181        Adds a reaction and its associated reactants and products into the
182        Network.
183
184        Parameters
185        ----------
186        reaction: NorthNet.Classes.Reaction
187            reaction to be added
188
189        Returns
190        -------
191        None
192        """
193        assert isinstance(
194            reaction, Classes.Reaction
195        ), """Network.add_reaction:
196            reaction arg should be a NorthNet Reaction object."""
197
198        if reaction.ReactionSMILES not in self.NetworkReactions:
199            reaction_smiles = reaction.ReactionSMILES
200
201            self.NetworkReactions[reaction_smiles] = reaction
202
203            for reactant in reaction.Reactants:
204                if reactant not in self.NetworkCompounds:
205                    new_compound = Classes.Compound(reactant)
206                    self.add_compound(new_compound)
207                if reaction_smiles not in self.NetworkCompounds[reactant].Out:
208                    self.NetworkCompounds[reactant].Out.append(reaction_smiles)
209
210            for product in reaction.Products:
211                if product not in self.NetworkCompounds:
212                    new_compound = Classes.Compound(product)
213                    self.add_compound(new_compound)
214                if reaction_smiles not in self.NetworkCompounds[product].In:
215                    self.NetworkCompounds[product].In.append(reaction_smiles)

248        Remove a list of reactions from the Network.

250        Parameters
251        ----------
252        remove_reactions: list[NorthNet.Classes.Reaction] or list[str]
253            reactions to be removed (either in reaction SMILES format or as
254                NorthNet.Classes.Reaction).
255
256        Returns
257        -------
258        None
259        """
260
261        assertion_msg = """Network.remove_reactions():
262            remove_reactions arg should be list of NorthNet Reaction objects"""
263
264        if isinstance(remove_reactions, list):
265            check_rxns = [isinstance(c, Classes.Reaction) for c in remove_reactions]
266            check_for_strings = [isinstance(r, str) for r in remove_reactions]
267
268            if not all(check_rxns) and not all(check_for_strings):
269                assert all(check_rxns), assertion_msg
270
271        else:
272            assert isinstance(remove_reactions, list), assertion_msg
273
274        for reaction in remove_reactions:
275            if isinstance(reaction, str):
276                r_key = reaction
277            else:
278                r_key = reaction.ReactionSMILES
279
280            for reactant in set(self.NetworkReactions[r_key].Reactants):
281                self.NetworkCompounds[reactant].Out.remove(r_key)
282            for product in set(self.NetworkReactions[r_key].Products):
283                self.NetworkCompounds[product].In.remove(r_key)
284
285            del self.NetworkReactions[r_key]

573    def convert_to_networkx(self):
574        """
575        Converts NorthNet network object to networkx object.
576
577        Parameters
578        ----------
579
580        Returns
581        -------
582        G: networkx.DiGraph
583            Networkx version of the NorthNet network.
584        """
585        import networkx as nx
586
587        Graph = nx.DiGraph()
588
589        for node in self.NetworkCompounds:
590            Graph.add_node(node)
591
592        for reaction in self.NetworkReactions:
593            for reactant in self.NetworkReactions[reaction].Reactants:
594                for product in self.NetworkReactions[reaction].Products:
595                    Graph.add_edge(reactant, reaction)
596                    Graph.add_edge(reaction, product)
597
598        return Graph

575        """
576        Converts NorthNet network object to networkx object.
577
578        Parameters
579        ----------
580
581        Returns
582        -------
583        G: networkx.DiGraph
584            Networkx version of the NorthNet network.
585        """
586        import networkx as nx
587
588        Graph = nx.DiGraph()
589
590        for node in self.NetworkCompounds:
591            Graph.add_node(node)
592
593        for reaction in self.NetworkReactions:
594            for reactant in self.NetworkReactions[reaction].Reactants:
595                for product in self.NetworkReactions[reaction].Products:
596                    Graph.add_edge(reactant, reaction)
597                    Graph.add_edge(reaction, product)
598
599        return Graph
